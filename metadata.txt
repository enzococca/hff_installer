version=1.0.0
author=ENZO COCCA
email=enzo.ccc@gmail.com

about=This plugin lets you install or update HFF directly from GitHub.
    It always installs the stable master branch (no development branches).
    The plugin automatically handles existing installations, removing old versions and
    ensuring the plugin folder is correctly named 'HFF' (required because v11.8+
    uses relative imports that fail if the folder is renamed by GitHub's zipball,
    e.g. 'HFF-master' or 'HFF-main').
    Compatible with both QGIS 3.x (Qt5) and QGIS 4.x (Qt6).

tracker=https://github.com/enzococca/HFF/issues
repository=https://github.com/enzococca/hff_installer

changelog=
    1.0.0 - Initial release
        - Download HFF master branch from enzococca/HFF
        - Automatic version detection from metadata.txt
        - Remove existing installations (HFF, HFF-master, HFF-main)
        - Rename installed folder to 'HFF'
        - Qt5/Qt6 compatibility
